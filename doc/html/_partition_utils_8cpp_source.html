250 void PartitionUtils::createPartition(const char* path, uint64_t size, int file_system){
251  if(!DiskUtils::getInstance().verifyDisk(path)){
252  return;
253  }
254  auto fd = open(path, O_RDWR, S_IRUSR | S_IWUSR);
255  if(-1 == fd){
256  std::cout << utils::colors::RED << "Failed to create new partition: open virtual disk file" << utils::colors::RESET << std::endl;
257  return;
258  }
259 
260  uint8_t* data_m = static_cast<uint8_t*>(mmap(nullptr, 0x1FF, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0));
261  if(MAP_FAILED == data_m){
262  std::cout << utils::colors::RED << "Failed to create new partition: maping failed" << utils::colors::RESET << std::endl;
263  close(fd);
264  return;
265  }
266 
267  close(fd);
268 
269  int index = 0;
270  uint64_t start;
271  for(; index < 4; index++){
272  if(data_m[0x1BE + (index * 16)] == 0x00){
273  break;
274  }
275  }
276  start = 512 + (index * 32);
277  uint64_t end = start + size;
278  // Write primary partition flag and filesystem type.
279  // Note an partition entry is exact 16 Bytes
280  data_m[0x1BE + (index * 16)] = static_cast<uint8_t>(0x80);
281  data_m[0x1BF + (index * 16)] = static_cast<uint8_t>(file_system);
282 
283  // Ill assign the LSB to the smallest position and the HSB to the highest position.
284  // Start address of partition
285  data_m[0x1C0 + (index * 16)] = static_cast<uint8_t>(start & 0xFF);
286  data_m[0x1C1 + (index * 16)] = static_cast<uint8_t>((start >> 8) & 0xFF);
287  data_m[0x1C2 + (index * 16)] = static_cast<uint8_t>((start >> 16) & 0xFF);
288  data_m[0x1C3 + (index * 16)] = static_cast<uint8_t>((start >> 24) & 0xFF);
289 
290  // End address of partition
291  data_m[0x1C4 + (index * 16)] = static_cast<uint8_t>(end & 0xFF);
292  data_m[0x1C5 + (index * 16)] = static_cast<uint8_t>((end >> 8) & 0xFF);
293  data_m[0x1C6 + (index * 16)] = static_cast<uint8_t>((end >> 16) & 0xFF);
294  data_m[0x1C7 + (index * 16)] = static_cast<uint8_t>((end >> 24) & 0xFF);
295 
296  uint16_t block_size = 4096;
297  uint64_t cluster_size = (size - block_size) / block_size;
298 
299  data_m[0x1C8 + (index * 16)] = static_cast<uint8_t>(cluster_size & 0xFF);
300  data_m[0x1C9 + (index * 16)] = static_cast<uint8_t>((cluster_size >> 8) & 0xFF);
301  data_m[0x1CA + (index * 16)] = static_cast<uint8_t>((cluster_size >> 16) & 0xFF);
302  data_m[0x1CB + (index * 16)] = static_cast<uint8_t>((cluster_size >> 24) & 0xFF);
303  data_m[0x1CC + (index * 16)] = static_cast<uint8_t>(block_size & 0xFF);
304  data_m[0x1CD + (index * 16)] = static_cast<uint8_t>((block_size >> 8) & 0xFF);
305 
306  munmap(data_m, 0x1FF);
307 
308  if(file_system == FAT || file_system == FAT_HIDDEN){
309  createFAT(path);
310  }
311 }
